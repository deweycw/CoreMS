 20    def get_mass_spectrum(self, loadSettings:bool =True) -> MassSpecCentroid:
 21        """
 22        Get the mass spectrum object from the processed mass list data.
 23
 24        Parameters
 25        ----------
 26        loadSettings : bool, optional
 27            Whether to load the settings for the mass spectrum. Default is True.
 28
 29        Returns
 30        -------
 31        MassSpecCentroid
 32            The mass spectrum object.
 33
 34        Raises
 35        ------
 36        ValueError
 37            If the input file is not a valid CoreMS file.
 38        """
 39
 40        dataframe = self.get_dataframe()
 41
 42        if not set(['H/C', 'O/C', 'Heteroatom Class', 'Ion Type', 'Is Isotopologue']).issubset(dataframe.columns):
 43            raise ValueError("%s it is not a valid CoreMS file" % str(self.file_location))
 44
 45        self.check_columns(dataframe.columns)
 46
 47        dataframe.rename(columns=self.parameters.header_translate, inplace=True)
 48
 49        polarity = dataframe['Ion Charge'].values[0]
 50
 51        output_parameters = self.get_output_parameters(polarity)
 52
 53        mass_spec_obj = MassSpecCentroid(dataframe.to_dict(orient='list'), output_parameters)
 54
 55        if loadSettings is True:
 56            self.load_settings(mass_spec_obj, output_parameters)
 57
 58        self.add_molecular_formula(mass_spec_obj, dataframe)
 59
 60        return mass_spec_obj

 62    def add_molecular_formula(self, mass_spec_obj, dataframe):
 63        """
 64        Add molecular formula information to the mass spectrum object.
 65
 66        Parameters
 67        ----------
 68        mass_spec_obj : MassSpecCentroid
 69            The mass spectrum object to add the molecular formula to.
 70        dataframe : pandas.DataFrame
 71            The processed mass list data.
 72            
 73        """
 74
 75        # check if is coreMS file
 76        if 'Is Isotopologue' in dataframe:
 77
 78            mz_exp_df = dataframe[Labels.mz].astype(float)
 79            formula_df = dataframe[dataframe.columns.intersection(Atoms.atoms_order)].copy()
 80            formula_df.fillna(0, inplace=True)
 81            formula_df.replace(b'nan', 0, inplace=True)
 82
 83            ion_type_df = dataframe["Ion Type"]
 84            ion_charge_df = dataframe["Ion Charge"]
 85            is_isotopologue_df = dataframe['Is Isotopologue']
 86            if 'Adduct' in dataframe:
 87                adduct_df = dataframe['Adduct']
 88            else:
 89                adduct_df = None
 90
 91        mass_spec_mz_exp_list = mass_spec_obj.mz_exp
 92
 93        for df_index, mz_exp in enumerate(mz_exp_df):
 94
 95            counts = 0
 96
 97            ms_peak_index = list(mass_spec_mz_exp_list).index(float(mz_exp))
 98
 99            if 'Is Isotopologue' in dataframe:
100
101                atoms = list(formula_df.columns.astype(str))
102                counts = list(formula_df.iloc[df_index].astype(int))
103
104                formula_list = [sub[item] for item in range(len(atoms))
105                                for sub in [atoms, counts]]
106            if sum(counts) > 0:
107
108                ion_type = str(Labels.ion_type_translate.get(ion_type_df[df_index]))
109                if adduct_df is not None:
110                    adduct_atom = str(adduct_df[df_index])
111                else:
112                    adduct_atom = None
113                mfobj = MolecularFormula(formula_list, int(ion_charge_df[df_index]), mspeak_parent=mass_spec_obj[ms_peak_index] , ion_type=ion_type, adduct_atom=adduct_atom)
114                mfobj.is_isotopologue = bool(is_isotopologue_df[df_index])
115                mass_spec_obj[ms_peak_index].add_molecular_formula(mfobj)
