356        """Get statistics about the GCMS data.
357
358        Parameters:
359        ----------
360        gcms : object
361            The low resolution GCMS object.
362
363        Returns:
364        -------
365        dict
366            A dictionary containing the data statistics.
367        """
368
369        matched_peaks = gcms.matched_peaks
370        no_matched_peaks = gcms.no_matched_peaks
371        unique_metabolites = gcms.unique_metabolites
372
373        peak_matchs_above_0p85 = 0
374        unique_peak_match_above_0p85 = 0
375        for match_peak in matched_peaks:
376            gc_peak_above_85 = 0
377            matches_above_85 = list(filter(lambda m: m.similarity_score >= 0.85, match_peak))
378            if matches_above_85:
379                peak_matchs_above_0p85 +=1
380            if len(matches_above_85) == 1:
381                unique_peak_match_above_0p85 += 1
382
383        data_stats = {}
384        data_stats['average_signal_noise'] = "ni"
385        data_stats['chromatogram_dynamic_range'] = gcms.dynamic_range
386        data_stats['total_number_peaks'] = len(gcms)
387        data_stats['total_peaks_matched'] = len(matched_peaks)
388        data_stats['total_peaks_without_matches'] = len(no_matched_peaks)
389        data_stats['total_matches_above_similarity_score_0.85'] = peak_matchs_above_0p85
390        data_stats['single_matches_above_similarity_score_0.85'] = unique_peak_match_above_0p85
391        data_stats['unique_metabolites'] = len(unique_metabolites)
392
393        return data_stats
